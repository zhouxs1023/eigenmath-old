Negate the following expressions.

\bigskip
{\bf a.} $p\to(q\land{\sim}\,r)$
$$\eqalign{
{\sim}\,(p\to(q\land{\sim}\,r))&\equiv{\sim}\,({\sim}\,p\lor(q\land{\sim}\,r))\cr
&\equiv p\land{\sim}\,(q\land{\sim}\,r)\cr
&\equiv p\land({\sim}\,q\lor r)\cr
}$$

\bigskip
{\bf b.} ${\sim}\,p\to(q\lor r)$
$$\eqalign{
{\sim}\,({\sim}\,p\to(q\lor r))&\equiv{\sim}\,(p\lor(q\lor r))\cr
&\equiv{\sim}\,p\land{\sim}\,(q\lor r)\cr
&\equiv{\sim}\,p\land{\sim}\,q\land{\sim}\,r
}$$

\bigskip
{\bf c.} $p\land(q\to r)$
$$\eqalign{
{\sim}\,(p\land(q\to r))&\equiv{\sim}\,p\lor{\sim}\,({\sim}\,q\lor r)\cr
&\equiv{\sim}\,p\lor(q\land{\sim}\,r)\cr
}$$
